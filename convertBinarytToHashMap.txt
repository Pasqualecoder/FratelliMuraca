// Supponendo che tu abbia recuperato il dato binario dal database come un array di byte
        byte[] binaryData = ...;

        // Deserializzazione del dato binario
        HashMap<String, Integer> hashMap = null;
        try (ByteArrayInputStream bais = new ByteArrayInputStream(binaryData);
             ObjectInputStream ois = new ObjectInputStream(bais)) {
            // Leggi l'oggetto dalla stream di input
            hashMap = (HashMap<String, Integer>) ois.readObject();
        } catch (IOException | ClassNotFoundException e) {
            e.printStackTrace();
        }

        // Ora hashMap contiene la `HashMap` deserializzata
        System.out.println("HashMap deserializzata: " + hashMap);
